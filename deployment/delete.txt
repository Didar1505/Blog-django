# Stop the services
sudo systemctl stop gunicorn.socket gunicorn.service

# Disable them so they don't start on reboot
sudo systemctl disable gunicorn.socket gunicorn.service

# Remove the systemd service files
sudo rm /etc/systemd/system/gunicorn.service
sudo rm /etc/systemd/system/gunicorn.socket

# Reload systemd to apply the removal
sudo systemctl daemon-reload
sudo systemctl reset-failed


# Remove the enabled link
sudo rm /etc/nginx/sites-enabled/PROJECT_NAME

# Remove the actual config file
sudo rm /etc/nginx/sites-available/PROJECT_NAME

# (Optional) If Certbot created a backup config, remove it too
# sudo rm /etc/nginx/sites-available/PROJECT_NAME-le-ssl.conf

# Restart Nginx to finalize changes
sudo nginx -t && sudo systemctl restart nginx
